cost = {
	trigger = {
		exists = owner
		owner = {
			OR = {
				is_lithoid_empire = yes
				AND = {
					is_hive_empire = no
					is_civic_plant_organic = no
				}
			}
		}
	}
	minerals = @berg_$category$_$building$_cost_t$tier$
}
cost = {
	trigger = {
		exists = owner
		owner = {
			NOT = { has_valid_civic = civic_integrated_bioarchitecture }
			OR = {
				is_hive_empire = yes
				is_civic_plant_organic = yes
			}
			is_lithoid_empire = no
		}
	}
	minerals = @berg_$category$_$building$_costTwoThird_t$tier$
	food = @berg_$category$_$building$_costOneThird_t$tier$
}
cost = {
	trigger = {
		exists = owner
		owner = {
			has_valid_civic = civic_integrated_bioarchitecture
			is_lithoid_empire = no
		}
	}
	food = @berg_$category$_$building$_cost_t$tier$
}
upkeep = {
	trigger = {
		exists = owner
		owner = {
			is_lithoid_empire = no
			OR = {
				is_hive_empire = yes
				is_civic_plant_organic = yes
			}
		}
	}
	energy = @berg_$category$_$building$_upkeepTwoThird_t$tier$
	food = @berg_$category$_$building$_upkeepOneThird_t$tier$
}
upkeep = {
	trigger = {
		exists = owner
		owner = {
			OR = {
				is_lithoid_empire = yes
				AND = {
					is_hive_empire = no
					is_civic_plant_organic = no
				}
			}
			NOT = { has_valid_civic = civic_integrated_bioarchitecture }
		}
	}
	energy = @berg_$category$_$building$_upkeep_t$tier$
}
