#PRIEST
triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			NOR = {
				has_civic = civic_defenders_faith
				has_valid_civic = civic_machine_cult
			}
		}
		$trigger$
	}
	modifier = {
		job_priest_add = $AMOUNT$
	}
}
triggered_desc = {
	trigger = {
		exists = owner
		owner = {
			NOR = {
				has_civic = civic_defenders_faith
				has_valid_civic = civic_machine_cult
			}
		}
		check_variable_arithmetic = {
			which = $AMOUNT$
			value > 0
		}
	}
	text = job_priest_effect_desc
}
