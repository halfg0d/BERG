		exists = owner
		owner = {
			NOR = {
				is_megacorp = yes
				AND = {
					berg_has_ecc_mod = yes
					has_civic = civic_defenders_faith
				}
			}
		}
		$trigger$
	}
	modifier = {
		job_priest_add = $AMOUNT$
	}
}
