triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = { NOT = { has_civic = civic_defenders_faith } }
		$trigger$
	}
	modifier = {
		job_priest_add = $AMOUNT$
	}
}

	trigger = {
		exists = owner
		owner = { NOT = { has_civic = civic_defenders_faith } }
		check_variable_arithmetic = {
			which = $AMOUNT$
			value > 0
		}
	}
	text = job_priest_effect_desc
}
