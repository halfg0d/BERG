	}

	}

	inline_script = {
		script = vanilla/misc/berg_branch_office_add
		branch_add = 0 
		branch_add = 0 
		osa_branch_add = 1
	}
	inline_script = {
		script = OST/ost_capital_colony_jobs
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { NOT = { has_active_tradition = tr_synthetics_finish } }
		}
		job_colonist_add = 2
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_active_tradition = tr_synthetics_finish }
		}
		job_roboticist_add = 1
		job_colonist_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_active_tradition = tr_cybernetics_organ_harvesting }
			has_current_purge = yes
			growing_species = { has_trait = trait_cybernetic }
		}
		pop_growth_speed = 0.10
	}

		potential = {
			is_capital = yes
			owner = {
				has_councilor = { COUNCILOR = "councilor_feudal_realm" } 
			}
		}
		country_naval_cap_add = 1
		mult = value:naval_cap_vassals_councilor_feudal_realm
	}

	resources = {
		category = planet_buildings

			energy = 1
		}

			trigger = {
				berg_has_osa_mod = yes
			}

		}
	}

	upgrades = {
		building_capital
	}
