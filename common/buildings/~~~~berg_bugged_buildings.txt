building_bugged_planetary_stock_exchange = {
	base_buildtime = 480
	base_cap_amount = 1

	category = trade

	potential = {
		has_bugged_mod = yes
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
	}

	allow = {
		has_major_upgraded_capital = yes
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_galactic_stock_exchange
						is_starbase_building_building = ring_galactic_stock_exchange
					}
				}
			}
		}
	}
	convert_to = {
		building_galactic_stock_exchange
	}
	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			has_modifier = resort_colony
			has_bugged_mod = no
		}
	}

	inline_script = {
		script = BUG/berg_galactic_stock_script
		tier = 2
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 600
			rare_crystals = 100
		}
		upkeep = {
			energy = 5
			rare_crystals = 2
		}
	}

	prerequisites = {
		"tech_galactic_markets"
	}

	upgrades = {
		"building_galactic_stock_exchange"
	}
	triggered_desc = {
		text = building_bugged_planetary_stock_exchange_effect_desc
	}

	ai_resource_production = {
		energy = 3
		trigger = {
			always = yes
		}
	}
	ai_resource_production = {
		energy = 5
		trigger = {
			solar_system = {
				has_system_trade_value >= 50
			}
		}
	}
	ai_resource_production = {
		energy = 3
		trigger = {
			solar_system = {
				has_system_trade_value >= 80
			}
		}
	}
}
