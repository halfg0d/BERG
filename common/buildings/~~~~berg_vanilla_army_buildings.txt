building_military_academy = {
	base_buildtime = @b2_time
	base_cap_amount = 1
	can_be_ruined = no
	category = army

	potential = {
		exists = owner
		owner = { berg_reanimator_civic = no }
	}

	allow = {
		has_upgraded_capital = yes
		if = {
			limit = {
				exists = owner
				owner = { is_ai = yes }
			}
			OR = {
				has_building = building_stronghold
				has_building = building_fortress
			}
		}
	}

	destroy_trigger = {
		exists = owner
		owner = {
			OR = {
				berg_reanimator_civic = yes
				AND = {
					is_ai = yes
					prev = {
						NOR = {
							has_building = building_stronghold
							has_building = building_fortress
						}
					}
				}
			}
		}
	}

	convert_to = {
		building_dread_encampment
	}
	inline_script = {
		script = vanilla/army/berg_academy_script
		tier = 1
	}
	inline_script = {
		script = building/resources/vanilla_military_academy
	}

	prerequisites = {
		"tech_centralized_command"
	}

	show_tech_unlock_if = {
		berg_reanimator_civic = no
	}
	upgrades = {
		"building_eutab_battle_praxeum" # EUTAB
		"building_mechanicum_academy" #OSA
		"building_avb_military_academy_2" #AVB
	}
}

building_planetary_shield_generator = {
	base_buildtime = @b3_time
	base_cap_amount = 1
	position_priority = 100
	can_be_ruined = no
	category = army

	allow = {
		has_upgraded_capital = yes
		hidden_trigger = { owner = { is_ai = no } }
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_orbital_shield_generator
						is_starbase_building_building = ring_orbital_shield_generator
					}
				}
			}
		}
	}
	upgrades = {
		"building_mem_planetary_shield_generator" #OSA
		"building_avb_planetary_shield_generator_2" #AVB
	}
	potential = {
		OR = {
			has_osa_mod = no
			AND = {
				has_osa_mod = yes
				NOR = {
					has_building = "building_mem_planetary_shield_generator"
					has_building = "building_mem_planetary_shield_generator_heavy"
					has_building = "building_mem_planetary_shield_generator_orila"
				}
				planet = {
					NOR = {
						uses_habitat_capitals = yes
						is_planet_class = pc_habitat
					}	
				}
			}
		}
	}
	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
	}
	inline_script = {
		script = vanilla/army/berg_shield_script
		tier = 1
	}
	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
		}
		upkeep = {
			energy = @b2_upkeep
		}
		inline_script = {
			script = ECC/berg_ecc_shield_script
			tier = 1
		}
		#OSA
		upkeep = {
			trigger = {
				has_osa_mod = yes
			}
			energy = 3
			minerals = 10
		}
	}

	prerequisites = {
		"tech_planetary_shield_generator"
	}
}
