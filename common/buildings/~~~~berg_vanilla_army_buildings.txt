building_planetary_shield_generator = {
	base_buildtime = @b3_time
	base_cap_amount = 1
	position_priority = 100
	can_be_ruined = no
	category = army

	allow = {
		has_upgraded_capital = yes
		hidden_trigger = { owner = { is_ai = no } }
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_orbital_shield_generator
						is_starbase_building_building = ring_orbital_shield_generator
					}
				}
			}
		}
	}
	upgrades = {
		"building_mem_planetary_shield_generator" #OSA
		"building_avb_planetary_shield_generator_2" #AVB
	}
	potential = {
		OR = {
			has_osa_mod = no
			AND = {
				has_osa_mod = yes
				NOR = {
					has_building = "building_mem_planetary_shield_generator"
					has_building = "building_mem_planetary_shield_generator_heavy"
					has_building = "building_mem_planetary_shield_generator_orila"
				}
				planet = {
					NOR = {
						uses_habitat_capitals = yes
						is_planet_class = pc_habitat
					}	
				}
			}
		}
	}
	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
	}
	inline_script = {
		script = vanilla/army/berg_shield_script
		tier = 1
	}
	inline_script = {
		script = building_resources/vanilla_shield
		tier = 1
	}

	prerequisites = {
		"tech_planetary_shield_generator"
	}
}
