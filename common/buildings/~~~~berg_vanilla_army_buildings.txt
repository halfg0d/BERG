building_military_academy = {
	base_buildtime = @b2_time
	base_cap_amount = 1
	can_be_ruined = no
	category = army

	potential = {
		exists = owner
		owner = { NOT = { has_valid_civic = civic_reanimated_armies } }
	}

	allow = {
		has_upgraded_capital = yes
		if = {
			limit = {
				exists = owner
				owner = { is_ai = yes }
			}
			OR = {
				has_building = building_stronghold
				has_building = building_fortress
			}
		}
	}

	destroy_trigger = {
		exists = owner
		owner = {
			OR = {
				has_valid_civic = civic_reanimated_armies
				AND = {
					is_ai = yes
					prev = {
						NOR = {
							has_building = building_stronghold
							has_building = building_fortress
						}
					}
				}
			}
		}
	}

	convert_to = {
		building_dread_encampment
	}
	inline_script = {
		script = vanilla/army/berg_academy_script
		tier = 1
	}
	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
		produces = {
			trigger = {
				berg_has_ethics_and_civic_classic_mod = yes
				owner = { has_civic = civic_martial_brotherhood }
			}
			unity = @berg_ecc_l1_unity
		}
	}

	prerequisites = {
		"tech_centralized_command"
	}

	show_tech_unlock_if = {
		NOT = { has_valid_civic = civic_reanimated_armies }
	}
}

building_planetary_shield_generator = {
	base_buildtime = @b3_time
	base_cap_amount = 1
	position_priority = 100
	can_be_ruined = no
	category = army

	allow = {
		has_upgraded_capital = yes
		hidden_trigger = { owner = { is_ai = no } }
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_orbital_shield_generator
						is_starbase_building_building = ring_orbital_shield_generator
					}
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
	}
	inline_script = {
		script = vanilla/army/berg_shield_script
		tier = 1

		}

			energy = @b2_upkeep
		}
