			berg_is_resort_world = yes

			AND = {
				owner = {
					is_ai = yes
				}
				has_any_industry_district = no
			}
		}
	}

	allow = {
		has_major_upgraded_capital = yes

		is_regular_empire = yes
	}

	ai_resource_production = {
		alloys = 6
		trigger = {
			planet_resource_compare = {
				resource = alloys
				value >= 40
				type = produces
			}
		}
