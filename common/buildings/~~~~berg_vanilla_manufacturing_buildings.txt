building_factory_1 = {
	base_buildtime = @b1_time
	base_cap_amount = 1
	category = manufacturing

	potential = {
		exists = owner
		owner = {
			OR = {
				country_uses_consumer_goods = yes
				berg_is_fear_of_the_dark_empire = yes
			}
		}
		NOR = {
			has_modifier = resort_colony
			has_modifier = crucible_colony
		}
		OR = {
			owner = {
				is_ai = no
			}
			and = {
				has_any_industry_district = yes
				owner = {
					should_ai_focus_on_trade = no
				}
			}
		}
		#zExpanded
		berg_slave_colony_manufactury = yes
	}

	destroy_trigger = {
		OR = {
			has_modifier = resort_colony
			has_modifier = crucible_colony

			AND = {
				exists = owner
				OR = {
					owner = { 
						AND = {
							country_uses_consumer_goods = no
							berg_is_fear_of_the_dark_empire = no
						}
					}
					AND = {
						owner = {
							is_ai = yes
						}
						has_any_industry_district = no
					}
					AND = {
						owner = {
							should_ai_focus_on_trade = yes
							has_monthly_income = { resource = consumer_goods value > 30 }
						}
					}
				}
			}
			berg_slave_colony_manufactury_not = yes
		}
	}
	inline_script = {
		script = vanilla/manufacturing/berg_factory_script
		tier = 1
	}
	inline_script = {
		script = building_resources/vanilla/factory
		tier = 1
	}

	prerequisites = {
		tech_basic_industry
	}

	upgrades = {
		building_factory_2
	}
}
