building_autochthon_monument = {
	base_buildtime = @b1_time
	base_cap_amount = 1
	category = unity

	potential = {
		exists = owner
		NOT = { has_modifier = slave_colony }
		owner = {
			NOR = {
				is_gestalt = yes
				is_megacorp = yes
				is_memorialist_empire = yes

	destroy_trigger = {
		exists = owner
		OR = {
			has_modifier = slave_colony
			owner = {
				OR = {
					is_gestalt = yes
					is_megacorp = yes
					is_memorialist_empire = yes
				}
			}
		}
	}

	inline_script = {
		script = building_resources/vanilla_monument_regular
		tier = 1
	}
	inline_script = {
		script = vanilla/unity/berg_regular_monument_script
		tier = 1
	}
	triggered_desc = {
		text = building_autochthon_monument_effect
	}
	prerequisites = {
		"tech_cultural_heritage"
	}

	upgrades = {
		building_heritage_site
	}

	convert_to = {
		building_corporate_monument
		building_simulation_1
		building_galactic_memorial_1
		building_sensorium_1
	}

	show_tech_unlock_if = {
		is_gestalt = no
		is_megacorp = no
		is_memorialist_empire = no
	}
}
